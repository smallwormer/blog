# /bin/bash
slab=$(cat /proc/meminfo|grep Slab| awk '{print $2}')
pagetables=$(cat /proc/meminfo | grep PageTables| awk '{print $2}')
kernelstack=$(cat /proc/meminfo | grep KernelStack| awk '{print $2}')
vmalloc=$(grep vmalloc /proc/vmallocinfo | awk '{total+=$2}; END {print total}')
let "vmalloc=$vmalloc/1000"

let "total=$slab"
let "total+=$pagetables"
let "total+=$kernelstack"
let "total+=$vmalloc"

echo "Kernel memory stats(KB):"
echo "Slab: $slab, Page tables: $pagetables, Kernel Stack: $kernelstack, vmalloc: $vmalloc"
echo "Total: $total"
echo
